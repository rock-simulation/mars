146  void ControllerConfigGUI::timerEvent(QTimerEvent* event) {
147  (void) event;
148  std::vector<interfaces::core_objects_exchange> objectList;
149  std::vector<interfaces::core_objects_exchange>::iterator iter;
150  std::vector<ControllerElem*>::iterator jter;
151  std::vector<ControllerElem*> addList;
152  std::vector<ControllerElem*> deleteList;
153  ControllerElem* newElem;
154  unsigned long current_id = controllers[controllerIDCombo->value().toInt()]->id;
155  int index = 1, switch_id = 0, count_delete = 0;
156  bool found, first = 1;
157  QStringList enumNames = controllerIDCombo->attributeValue("enumNames").toStringList();
158 
159  addList.push_back(controllerElem);
160 
161  control->controllers->getListController(&objectList);
162 
163  // first we delete the nodes that are not in the list
164  for(jter=controllers.begin(); jter!=controllers.end(); ++jter) {
165  if(first) first = 0;
166  else {
167  found = false;
168  for(iter=objectList.begin(); iter!=objectList.end(); ++iter) {
169  if((*jter)->id == (*iter).index) {
170  found = true;
171  (*jter)->index = index++;
172  addList.push_back((*jter));
173  objectList.erase(iter);
174  break;
175  }
176  }
177  if(!found) {
178  if((*jter)->id == current_id) switch_id = 1;
179  // wir brauchen eine delete list weil wir von hinten löschen müssen
180  enumNames.removeAt((*jter)->index - count_delete++);
181  controllerIDCombo->setAttribute("enumNames", enumNames);
182  delete (*jter);
183  }
184  }
185  }
186 
187  // then we add the nodes that are still in the object list
188  for(iter=objectList.begin(); iter!=objectList.end(); ++iter) {
189  newElem = new ControllerElem;
190  newElem->id = (*iter).index;
191  newElem->name = (*iter).name;
192  newElem->index = index++;
193  addList.push_back(newElem);
194  enumNames << QString::fromStdString((*iter).name);
195  controllerIDCombo->setAttribute("enumNames", enumNames);
196  }
197  objectList.clear();
198 
199  controllers = addList;
200  if(switch_id) controllerIDCombo->setValue(0);
201  }
