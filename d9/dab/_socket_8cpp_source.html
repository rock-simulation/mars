388  TCPBaseSocket* TCPBaseSocket::accept() {
389  if(sock_fd == INVALID_SOCKET) {
390  fprintf(stderr, "TCPBaseSocket::accept: invalid socket\n");
391  return NULL;
392  }
393  long now = 0, deadline = 0;
394  double timeout = sock_timeout;
395  bool has_timeout = sock_timeout > 0.;
396  if(has_timeout) {
397  now = mars::utils::getTime();
398  deadline = now + (long)(sock_timeout * 1000.);
399  }
400  while(true) {
401  errno = 0;
402  int selected = internal_select(0, timeout);
403  if(selected == 1) {
404  TCPBaseSocket *client = new TCPBaseSocket;
405  client->sock_fd = ::accept(sock_fd, NULL, NULL);
406  return client;
407  } else if(selected == 0) {
408  fprintf(stderr, "TCPBaseSocket::accept: select failed\n");
409  return NULL;
410  }
411  if(!has_timeout || (!CHECK_ERRNO(EWOULDBLOCK) && !CHECK_ERRNO(EAGAIN)))
412  break;
413  timeout = mars::utils::getTimeDiff(now) - deadline;
414  now = mars::utils::getTime();
415  }

497  int TCPBaseSocket::internal_select(bool writing, double timeout) {
498  if(sock_timeout <= 0.)
499  return 1;
500  if(!isConnected())
501  return 0;
502  if(timeout < 0.)
503  return 1;
504  struct timeval timeoutStruct;
505  timeoutStruct.tv_sec = (int)timeout;
506  timeoutStruct.tv_usec = (int)((timeout - timeoutStruct.tv_sec) * 1e6);
507  fd_set fds;
508  FD_ZERO(&fds);
509  FD_SET(sock_fd, &fds);
510  int n;
511  if(writing)
512  n = select((int)(sock_fd+1), NULL, &fds, NULL, &timeoutStruct);
513  else
514  n = select((int)(sock_fd+1), &fds, NULL, NULL, &timeoutStruct);
515  if(n < 0)
516  return -1;
517  if(n == 0)
518  return 1;
519  return 0;
520  }
