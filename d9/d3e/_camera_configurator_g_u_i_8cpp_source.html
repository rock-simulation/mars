102  void CameraConfiguratorGUI::valueChanged(QtProperty *property, const QVariant &value)
103  {
104  if (filled == false) return;
105 
106  if (property == winIDCombo)
107  cameraSelected(value.toInt());
108  else if (property == nodeIDCombo)
109  checkStateChanged(lockPos->value().toBool());
110  else if (property == lockRot)
111  checkRotationChanged(value.toBool());
112  else if (property == lockPos)
113  checkStateChanged(value.toBool());
114  else if (property == xPosBox || property == yPosBox || property == zPosBox)
115  setOffsetPos();
116  else if (property == xRotBox || property == yRotBox || property == zRotBox)
117  setOffsetRot();
118  else if (property == frustum)
119  checkFrustumChanged(value.toBool());
120  else if (property == frt_left || property == frt_right || property == frt_top ||
121  property == frt_bottom || property == frt_near || property == frt_far)
122  if(set_frustum)
123  setFrustum();
124 
125 
126  (void)value;
127  }

296  void CameraConfiguratorGUI::removeCamera(CameraConfig* camera) {
297  unsigned int i = 0;
298  std::vector<CameraConfig*>::iterator iter;
299  QStringList enumNames = winIDCombo->attributeValue("enumNames").toStringList();
300  bool found;
301

306  if(iter!=cameras.end()) {

321  void CameraConfiguratorGUI::timerEvent(QTimerEvent* event) {
322  (void) event;
323  std::vector<interfaces::core_objects_exchange> objectList;
324  std::vector<interfaces::core_objects_exchange>::iterator iter;
325  std::vector<NodeElem*>::iterator jter;
326  std::vector<NodeElem*> addList;
327  std::vector<NodeElem*> deleteList;
328  NodeElem* newElem;
329  unsigned long current_id = nodes[nodeIDCombo->value().toInt()]->id;
330  int index = 1, switch_id = 0, count_delete = 0;
331  bool found, first = 1;
332  QStringList enumNames = nodeIDCombo->attributeValue("enumNames").toStringList();
333 
334  addList.push_back(nodeElem);
335 
336  control->nodes->getListNodes(&objectList);
337 
338  // first we delete the nodes that are not in the list
339  for(jter=nodes.begin(); jter!=nodes.end(); ++jter) {
340  if(first) first = 0;
341  else {
342  found = false;
343  for(iter=objectList.begin(); iter!=objectList.end(); ++iter) {
344  if((*jter)->id == (*iter).index) {
345  found = true;
346  (*jter)->index = index++;
347  addList.push_back((*jter));
348  objectList.erase(iter);
349  break;
350  }
351  }
352  if(!found) {
353  if((*jter)->id == current_id) switch_id = 1;
354  // wir brauchen eine delete list weil wir von hinten lschen mssen
355  enumNames.removeAt((*jter)->index - count_delete++);
356  nodeIDCombo->setAttribute("enumNames", enumNames);
357  delete (*jter);
358  }
359  }
360  }
361 
362  // then we add the nodes that are still in the object list
363  for(iter=objectList.begin(); iter!=objectList.end(); ++iter) {
364  newElem = new NodeElem;
365  newElem->id = (*iter).index;
366  newElem->name = (*iter).name;
367  newElem->index = index++;
368  addList.push_back(newElem);
369  enumNames << QString::fromStdString((*iter).name);
370  nodeIDCombo->setAttribute("enumNames", enumNames);
371  }
372  objectList.clear();
373 
374  nodeMutex.lock();
375  nodes = addList;
376  if(switch_id) nodeIDCombo->setValue(0);
377  nodeMutex.unlock();
378  }
