187  void MultiResHeightMapRenderer::recalcSteps() {
188  highStepX = stepX = targetWidth / double(getLowResCellCntX());
189  highStepY = stepY = targetHeight / double(getLowResCellCntY());
190 
191  highWidth = highHeight = 1;
192  double radius = 0.05;
193  double desiredStep = radius * 0.2;
194 
195  while(highStepX > desiredStep) {
196  highStepX *= 0.5;
197  }
198  while(highStepY > desiredStep) {
199  highStepY *= 0.5;
200  }
201 
202  highWidth = stepX / highStepX;
203  highHeight = stepY / highStepY;
204 
205  dirty = true;
206  }

277  return false;
278  }

280  for(int x = 0; x < width; ++x) {
281  int index = y*width+x;
282  vertices[index].position[0] = x * stepX * scaleX;
283  vertices[index].position[1] = y * stepY * scaleY;
284  vertices[index].position[2] = heightData[y][x] * scaleZ;
285  vertices[index].texCoord[0] = x * stepX * scaleX*texScaleX;
286  vertices[index].texCoord[1] = y * stepY * scaleY*texScaleY;
287 
288  vertices[index].normal[0] = 0.0;
289  vertices[index].normal[1] = 0.0;
290  vertices[index].normal[2] = 1.0;
291  vertices[index].tangent[0] = 0.0;

296

300  free(vertices);

302  }
303  for(int y=0; y<height-1; ++y) {
304  for(int x=0; x<width-1; ++x) {
305  int indexOffset = y*(width-1)*6+x*6;
306  indices[indexOffset+0] = (y+1) * width + x;
307  indices[indexOffset+1] = y * width + x;
308  indices[indexOffset+2] = (y+1) * width + x+1;
309 
310  indices[indexOffset+3] = (y+1) * width + x+1;

312  indices[indexOffset+5] = y * width + x+1;

314  }
315

317  glBufferData(GL_ARRAY_BUFFER, numVertices * sizeof(VertexData), vertices,
318  GL_DYNAMIC_DRAW);
319 
320  // Transfer index data to VBO 1
321  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, vboIds[vboId+1]);
322  glBufferData(GL_ELEMENT_ARRAY_BUFFER, numIndices * sizeof(GLuint), indices,
323  GL_DYNAMIC_DRAW);
324 
325  if(highRes) {
326  highVertices = vertices;
327  highIndices = indices;
328  }
329  else {
330  this->vertices = vertices;
331  this->indices = indices;
332  }
333  return true;
334  }

466  double MultiResHeightMapRenderer::interpolateCell(int gridX, int gridY,
467  double x, double y) {
468  double dx = (x - gridX*stepX) / stepX;
469  double dy = (y - gridY*stepY) / stepY;
470  double cornerHeights[4];
471  assert((gridY < getLowResCellCntY()) && (gridX < getLowResCellCntX()));
472  cornerHeights[0] = heightData[gridY][gridX];
473  cornerHeights[1] = heightData[gridY+1][gridX];
474  cornerHeights[2] = heightData[gridY][gridX+1];
475  cornerHeights[3] = heightData[gridY+1][gridX+1];
476  double height = (cornerHeights[0] * (1-dx) * (1-dy) +
477  cornerHeights[1] * (1-dx) * dy +
478  cornerHeights[2] * dx * (1-dy) +
479  cornerHeights[3] * dx * dy);
480  return height;
481  }

484

500  index = x2 + iy*getHighResVertexCntX() + ix;
501  double x = tile->x*stepX + ix*highStepX;
502  double y = tile->y*stepY + iy*highStepY;
503  double z = interpolateCell(tile->x, tile->y, x, y);
504  tile->heightData[iy][ix] = z;
505  vertices[index].position[0] = x * scaleX;
506  vertices[index].position[1] = y * scaleY;
507  vertices[index].position[2] = z * scaleZ;
508  vertices[index].texCoord[0] = x * scaleX*texScaleX;
509  vertices[index].texCoord[1] = y * scaleY*texScaleY;
510  }
511  }
512  // calculate normals in a separate loop because the calculation depends on the neighbours
513  for(int iy = 0; iy < getHighResVertexCntY(); ++iy) {
514  for(int ix = 0; ix < getHighResVertexCntX(); ++ix) {
515  index = x2 + iy*getHighResVertexCntX() + ix;

521  }
522  }
523  }
524 
525  if(indices) {
526  int offset = tile->indicesArrayOffset;
527  int x2 = tile->verticesArrayOffset;
528 
529  for(int iy = 0; iy < getHighResCellCntY(); ++iy) {

531  int indexOffset = offset+iy*getHighResCellCntX()*6+ix*6;
532  indices[indexOffset+0] = x2+((iy+1)*getHighResVertexCntX())+ix;
533  indices[indexOffset+1] = x2+( iy *getHighResVertexCntX())+ix;
534  indices[indexOffset+2] = x2+((iy+1)*getHighResVertexCntX())+ix+1;
535 
536  indices[indexOffset+3] = x2+((iy+1)*getHighResVertexCntX())+ix+1;
537  indices[indexOffset+4] = x2+( iy *getHighResVertexCntX())+ix;
538  indices[indexOffset+5] = x2+( iy *getHighResVertexCntX())+ix+1;
539  }
540  }
541  }
542  highIndicesToDraw += getHighResCellCntY()*getHighResCellCntX()*6;

557  void MultiResHeightMapRenderer::collideSphereI(double xPos, double yPos,
558  double zPos, double radius) {
559 
560  if(!isInitialized) {
561  return;
562  }
563 
564  // first check if we are within the current heightmap
565  if((xPos < 0 || yPos < 0) ||
566  (xPos > targetWidth || yPos > targetHeight)) {
567  return;
568  }
569 
570  if(!highIsInitialized) highInitialize();
571  if(!highIsInitialized) return;
572 
573  std::list<SubTile*> toProcess;
574  std::list<SubTile*>::iterator iter;
575

864  void MultiResHeightMapRenderer::getNormal(int x, int y, int mx, int my,
865  double x_step, double y_step,
866  double **height_data,
867  float *normal,
868  float *tangent,
869  bool skipBorder) {
870  double vx1, vz1, vy2, vz2;
871 
872  if(skipBorder) {
873  if(x < mx-2 && x > 1) {
874  vz1 = height_data[y][x+1] - height_data[y][x-1];
875  vx1 = x_step*2.0;
876  }
877  else if(x==0) {
878  vz1 = height_data[y][x+2] - height_data[y][x+1];
879  vx1 = x_step;
880  }
881  else if(x==1) {
882  vz1 = height_data[y][x+1] - height_data[y][x];
883  vx1 = x_step;
884  }
885  else if(x==mx-1) {
886  vz1 = height_data[y][x-1] - height_data[y][x-2];
887  vx1 = x_step;
888  }
889  else {
890  vz1 = height_data[y][x] - height_data[y][x-1];
891  vx1 = x_step;
892  }
893 
894  if(y > 1 && y < my-2) {
895  vz2 = height_data[y+1][x] - height_data[y-1][x];
896  vy2 = y_step*2.0;
897  }
898  else if(y==0) {
899  vz2 = height_data[y+2][x] - height_data[y+1][x];
900  vy2 = y_step;
901  }
902  else if(y==1) {
903  vz2 = height_data[y+1][x] - height_data[y][x];
904  vy2 = y_step;
905  }
906  else if(y==my-1) {
907  vz2 = height_data[y-1][x] - height_data[y-2][x];
908  vy2 = y_step;
909  }
910  else {
911  vz2 = height_data[y][x] - height_data[y-1][x];
912  vy2 = y_step;
913  }
914  }
915  else {
916  if(x != 0 && x != mx-1) {
917  vz1 = height_data[y][x+1] - height_data[y][x-1];
918  vx1 = x_step*2.0;
919  }
920  else if(x==0) {
921  vz1 = height_data[y][x+1] - height_data[y][x];
922  vx1 = x_step;
923  }
924  else {
925  vz1 = height_data[y][x] - height_data[y][x-1];
926  vx1 = x_step;
927  }
928 
929  if(y != 0 && y != my-1) {
930  vz2 = height_data[y+1][x] - height_data[y-1][x];
931  vy2 = y_step*2.0;
932  }
933  else if(y==0) {
934  vz2 = height_data[y+1][x] - height_data[y][x];
935  vy2 = y_step;
936  }
937  else {
938  vz2 = height_data[y][x] - height_data[y-1][x];
939  vy2 = y_step;
940  }
941  }
942 
943  vz1 *= scaleZ;
944  vx1 *= scaleX;
945  vz2 *= scaleZ;
946  vy2 *= scaleY;
947 
948  normal[0] = -vz1*vy2;
949  normal[1] = -vz2*vx1;
950  normal[2] = vx1*vy2;
951  normalize(normal);
952  tangent[0] = vx1;
953  tangent[1] = vy2;
954  tangent[2] = vz1 + vz2;
955  tangent[3] = 0.0;
956  normalize(tangent);
