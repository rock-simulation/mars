347  bool SMURFLoader::loadFile(std::string filename, std::string tmpPath,
348  std::string robotname) {
349  LOG_INFO("urdf_loader: prepare loading");
350  entitylist.clear();
351 
352  // split up filename in path + _filename and retrieve file extension
353  std::string file_extension = utils::getFilenameSuffix(filename);
354  fprintf(stderr, "SMURFLoader::loadFile: file_extension=%s\n", file_extension.c_str());
355  std::string path = utils::getPathOfFile(filename);
356  std::string _filename = filename;
357  utils::removeFilenamePrefix(&_filename);
358 
359  // need to unzip into a temporary directory
360  if (file_extension == ".zsmurf" or file_extension == ".zsmurfs") {
361  if (unzip(tmpPath, filename) == 0) {
362  path = tmpPath;
363  return 0;
364  }
365  }
366 
367  // read in the provided file - .smurfs / .smurf / .urdf
368  configmaps::ConfigMap map;
369  std::string uri;
370  fprintf(stderr, "Reading in %s...\n", (path+_filename).c_str());
371  if(file_extension == ".smurfs") {
372  configmaps::ConfigVector::iterator it;
373  map = configmaps::ConfigMap::fromYamlFile(path+_filename, false);
374  map.toYamlFile("smurfs_debugmap.yml");
375  for (it = map["smurfs"].begin(); it != map["smurfs"].end(); ++it) {
376  uri = (std::string)(*it)["URI"];
377  (*it)["path"] = path+utils::getPathOfFile(uri);
378  utils::removeFilenamePrefix(&uri);
379  (*it)["URI"] = uri;
380  entitylist.push_back((*it).children);
381  }
382  } else if(file_extension == ".smurf") {
383  // if we have only one smurf, only one with rudimentary data is added to the smurf list
384  //map["URI"] = _filename;
385  // map = configmaps::ConfigMap::fromYamlFile(path+_filename, true);
386  map["path"] = path;
387  map["URI"] = _filename;
388  map["type"] = "smurf";
389  map["name"] = "";
390  entitylist.push_back(map);
391  } else if(file_extension == ".urdf") {
392  map["URI"] = _filename;
393  map["path"] = path;
394  map["name"] = "";
395  map["type"] = "urdf";
396  entitylist.push_back(map);
397  } else if(file_extension == ".svg") {
398  /* A smurf svg can contain multiple entities, thus we have to loop over them.
399  * Each svg object must contain a "file" property, specifying the yaml file in
400  * which the missing parameters are defined, so we have to append that.
401  */
402  std::vector<configmaps::ConfigMap> svg_entities;
403  parseSVG(&svg_entities, path+_filename);
404  for(std::vector<configmaps::ConfigMap>::iterator it = svg_entities.begin();
405  it!=svg_entities.end(); ++it) {
406  map = *it;
407  map.append(ConfigMap::fromYamlFile((std::string)map["file"], true)); //FIXME: path?
408  fprintf(stderr, "Loading config for svg entity: %s", ((std::string)map["file"]).c_str());
409  entitylist.push_back(map);
410  }
411  }
412 
413  //load assembled smurfs
414  fprintf(stderr, "Creating simulation entities...\n");
415  for (std::vector<configmaps::ConfigMap>::iterator sit = entitylist.begin();
416  sit != entitylist.end(); ++sit) {
417  configmaps::ConfigMap tmpmap;
418  tmpmap = *sit;
419  factoryManager->createEntity(tmpmap);
420  }
421 
422  return 1; //TODO: check number of successfully loaded entities before returning 1
423  }
